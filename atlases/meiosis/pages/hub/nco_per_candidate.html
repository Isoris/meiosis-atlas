What this page shows
spec · phase C
Sister page to crossovers_per_candidate, same per-inversion- candidate keying, but for the green/yellow side of the recombination palette: NCO (non-crossover recombination tracts) and gene conversion. Two views: an ideogram of tracts inside the candidate ± flank, and a tract-length / telomere-distance distribution.
NCO and GC are kept together because the same pedigree-detection pipeline emits both, distinguished only by the kind ∈ {nco, gc} field on each tract.
Migrated 2026-05-19 from genome-atlas page12. Now lives in the Meiosis Atlas next to its sister crossovers_per_candidate page.

Direct analogue of crossovers_per_candidate's View 2. Two curves on the same axes: NCO (green) and GC (yellow), each with a 95% CI band. Useful for spotting whether GC tracts cluster at telomeres (often yes) and whether NCO does (often less so).

Import schema — nco_gc_track.json (per inversion candidate)
contract
Same per-candidate keying as crossovers_per_candidate. Filename is data/nco_gc/<candidate_id>.json. Empty tracts is legal and just renders an empty ideogram / flat curves. 
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "title": "nco_gc_track (per inversion candidate)",
  "type": "object",
  "required": ["candidate_id", "candidate_span", "tracts"],
  "properties": {
    "candidate_id": {
      "type": "string",
      "description": "Inversion-candidate id (matches shared.candidate)."
    },
    "candidate_span": {
      "type": "object",
      "required": ["chrom", "start_bp", "end_bp"],
      "properties": {
        "chrom":    { "type": "string" },
        "start_bp": { "type": "integer", "minimum": 0 },
        "end_bp":   { "type": "integer", "minimum": 0 }
      }
    },
    "flank_bp": {
      "type": "integer",
      "minimum": 0,
      "default": 5000000
    },

    "cohort":      { "type": "string" },
    "individuals": { "type": "integer", "minimum": 1 },

    "chroms": {
      "type": "array",
      "items": {
        "type": "object",
        "required": ["id", "length_bp"],
        "properties": {
          "id":          { "type": "string" },
          "length_bp":   { "type": "integer", "minimum": 1 },
          "telomere_L":  { "type": "integer" },
          "telomere_R":  { "type": "integer" }
        }
      }
    },

    "tracts": {
      "type": "array",
      "description": "One entry per NCO or gene-conversion tract.",
      "items": {
        "type": "object",
        "required": ["chrom", "start_bp", "end_bp", "kind"],
        "properties": {
          "chrom":    { "type": "string" },
          "start_bp": { "type": "integer", "minimum": 0 },
          "end_bp":   { "type": "integer", "minimum": 0 },
          "kind":     { "enum": ["nco", "gc"] },
          "indiv":    { "type": "string" },
          "in_span":  { "type": "boolean" }
        }
      }
    },

    "telomere_curve": {
      "type": "object",
      "properties": {
        "bins": { "type": "array", "items": { "type": "number", "minimum": 0, "maximum": 1 } },
        "nco":  { "$ref": "#/$defs/curve" },
        "gc":   { "$ref": "#/$defs/curve" }
      }
    },
    "curve_params": {
      "type": "object",
      "properties": {
        "smoother":     { "enum": ["loess", "spline", "none"] },
        "span":         { "type": "number", "minimum": 0, "maximum": 1 },
        "ci_bootstrap": { "type": "integer", "minimum": 0 }
      }
    }
  },

  "$defs": {
    "curve": {
      "type": "object",
      "properties": {
        "y":     { "type": "array", "items": { "type": "number" } },
        "ci_lo": { "type": "array", "items": { "type": "number" } },
        "ci_hi": { "type": "array", "items": { "type": "number" } }
      }
    }
  }
}
